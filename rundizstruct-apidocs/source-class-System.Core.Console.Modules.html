120:     /**
121:      * Copy module's composer.json to app's composer.json
122:      * 
123:      * @param array $options The data that should contain module_system_name in array key.
124:      * @return boolean Return true on success written, false on failed.
125:      */
126:     private function copyModuleComposer(array $options)
127:     {
128:         $moduleComposerString = file_get_contents(MODULE_PATH . DIRECTORY_SEPARATOR . $options['module_system_name'] . DIRECTORY_SEPARATOR . 'composer.json');
129:         $rootComposerString = file_get_contents(ROOT_PATH . DIRECTORY_SEPARATOR . 'composer.json');
130:         $ModuleComposerObject = json_decode($moduleComposerString);
131:         $RootComposerObject = json_decode($rootComposerString);
132: 
133:         // begins add require and require-dev (if available) to root composer.json.
134:         if (isset($ModuleComposerObject->require) && is_object($ModuleComposerObject->require)) {
135:             foreach ($ModuleComposerObject->require as $name => $version) {
136:                 $RootComposerObject->require->{$name} = $version;
137:             }// endforeach;
138:             unset($name, $version);
139:         }
140:         if (isset($ModuleComposerObject->{'require-dev'}) && is_object($ModuleComposerObject->{'require-dev'})) {
141:             foreach ($ModuleComposerObject->{'require-dev'} as $name => $version) {
142:                 $RootComposerObject->{'require-dev'}->{$name} = $version;
143:             }// endforeach;
144:             unset($name, $version);
145:         }
146: 
147:         // finished add, write to json string with pretty print and then write into file.
148:         $newRootComposerString = json_encode($RootComposerObject, JSON_UNESCAPED_SLASHES|JSON_PRETTY_PRINT);
149:         $writeComposerResult = file_put_contents(ROOT_PATH . DIRECTORY_SEPARATOR . 'composer.json', $newRootComposerString);
150: 
151:         // clear
152:         unset($moduleComposerString, $ModuleComposerObject, $newRootComposerString, $rootComposerString, $RootComposerObject);
153: 
154:         // finished copy composer.json
155:         if ($writeComposerResult !== false) {
156:             return true;
157:         } else {
158:             return false;
159:         }
160:     }// copyModuleComposer

179:         // prepare options for modules table.

181:         $options['module_system_name'] = $input->getOption('module_system_name');
182:         $options['module_enable'] = '1';

184:         // validate module folder exists , Installer.php file maybe exists.
185:         $routes = null;
186:         $validated = $this->validateFolderAndFileExists($Io, $validated, $options, $routes);
187: 
188:         // validate the current module system name must not exists in the database.
189:         $ModulesDb = new \System\Core\Models\ModulesDb($this->Db);
190:         if (!empty($ModulesDb->get($options['module_system_name']))) {
191:             // if NOT empty module system name (module folder name) means exists.
192:             $Io->error('The module you entered is already added in the database.');
193:             $validated = false;
194:         }
195: 
196:         // validate duplicate route.
197:         if (isset($validated) && $validated === true && isset($routes) && is_array($routes)) {
198:             $RoutesDb = new \System\Core\Models\RoutesDb($this->Db);
199:             $routesDbArray = $RoutesDb->getRoutes();
200:             unset($RoutesDb);
201:             $Router = new \System\Libraries\Router($this->SlimApp);
202:             foreach ($routes as $key => $route) {
203:                 if (!isset($route['route_core']) || (isset($route['route_core']) && $route['route_core'] != '1')) {
204:                     $routes[$key]['route_core'] = '0';
205:                 }
206:                 if (!isset($route['route_order']) || (isset($route['route_order']) && !is_numeric($route['route_order']))) {
207:                     $routes[$key]['route_order'] = '99';
208:                 }
209:                 if (isset($route['route_group'])) {
210:                     // if this route is route group.
211:                     $routeUrlKey = key($route['route_group']);
212:                     if (is_array($routesDbArray) && array_key_exists($routeUrlKey, $routesDbArray)) {
213:                         $Io->error('The specified route group path is already exists or it may duplicate with other single route URL. ('.$routesDbArray[$routeUrlKey].').');
214:                         $validated = false;
215:                         break;
216:                     }
217:                 } else {
218:                     // if this route is not route group, it is a normal route.
219:                     if (!isset($route['route_method']) || (isset($route['route_method']) && $route['route_method'] != 'map' && !$Router->isAllowedMethods($route['route_method']))) {
220:                         // if no route_method or the route method is not allowed then set new one to "any".
221:                         $routes[$key]['route_method'] = 'any';
222:                     }
223:                     if (isset($route['route_method']) && $route['route_method'] == 'map') {
224:                         if (
225:                             !isset($route['route_map_method']) || 
226:                             (
227:                                 isset($route['route_map_method']) &&
228:                                 (
229:                                     !is_array($route['route_map_method']) ||
230:                                     (
231:                                         is_array($route['route_map_method']) &&
232:                                         !$Router->isAllowedMethods($route['route_map_method'])
233:                                     )
234:                                 )
235:                             )
236:                         ) {
237:                             // if route method is "map" but route map methods are incorrect or not exists then set new one to [GET, POST].
238:                             $routes[$key]['route_map_method'] = ['GET', 'POST'];
239:                         }
240:                     }
241:                     if (isset($route['route_uri']) && is_string($route['route_uri'])) {
242:                         // if route_uri was set.
243:                         $routeUrlKey = $route['route_uri'];
244:                         if (is_array($routesDbArray) && array_key_exists($routeUrlKey, $routesDbArray)) {
245:                             $Io->error('The specified route URL is already exists. ('.$routeUrlKey.').');
246:                             $validated = false;
247:                             break;
248:                         }
249:                     } else {
250:                         // if route_uri was NOT set.
251:                         $Io->error('There is no specify route URL in one of your routes in the module '.$options['module_system_name'].'.');
252:                         $validated = false;
253:                         break;
254:                     }
255:                     if (!isset($route['route_controller']) || (isset($route['route_controller']) && !is_string($route['route_controller']))) {
256:                         $Io->error('There is no specify route controller in one of your routes in the module '.$options['module_system_name'].'.');
257:                     }
258:                     if (isset($route['route_name']) && !is_string($route['route_name'])) {
259:                         $Io->error('Invalid route name type in one of your routes in the module '.$options['module_system_name'].'. The route name must be string.');
260:                     }
261:                 }
262:                 unset($routeUrlKey);
263:             }// endforeach;
264:             unset($key, $route, $Router, $routesDbArray);
265:         }
266: 
267:         if (isset($validated) && $validated === true) {
268:             // if validated & passed, try to call `install()` method on `Installer` class first.
269:             // copy assets folder to public/Modules/[module_system_name]/assets folder.
270:             if (is_dir(MODULE_PATH . DIRECTORY_SEPARATOR . $options['module_system_name'] . DIRECTORY_SEPARATOR . 'assets')) {
271:                 $Fs = new \System\Libraries\FileSystem(PUBLIC_PATH . DIRECTORY_SEPARATOR . 'Modules');
272:                 $Fs->copyFolderRecursive(MODULE_PATH . DIRECTORY_SEPARATOR . $options['module_system_name'] . DIRECTORY_SEPARATOR . 'assets', $options['module_system_name'] . DIRECTORY_SEPARATOR . 'assets');
273:                 unset($Fs);
274:             }
275: 
276:             // try to call `install()` method on `Installer` class.
277:             $Installer = $this->getModuleInstallerClass($options);
278:             if (is_object($Installer) && method_exists($Installer, 'install')) {
279:                 try {
280:                     $Installer->install($this->Db);
281:                 } catch (\Exception $e) {
282:                     // failed to call install() method, delete any copied folder.
283:                     $Fs = new \System\Libraries\FileSystem(PUBLIC_PATH . DIRECTORY_SEPARATOR . 'Modules');
284:                     $Fs->deleteFolder($options['module_system_name'], true);
285:                     unset($Fs);
286: 
287:                     $Io->error($e->getMessage());
288:                     die();
289:                 }
290:             }
291:             unset($Installer);
292: 
293:             // add a module to db.
294:             $addResult = $ModulesDb->add($options);
295: 
296:             if (isset($addResult) && is_array($addResult) && isset($addResult['result']) && isset($addResult['module_id']) && $addResult['result'] === true) {
297:                 // if added successfully.
298:                 if (isset($routes) && is_array($routes)) {
299:                     $RoutesDb = new \System\Core\Models\RoutesDb($this->Db);
300:                     $Command = $this->getApplication()->find('system\route');
301:                     foreach ($routes as $route) {
302:                         $route['module_id'] = $addResult['module_id'];
303:                         // call other command to add route.
304:                         // use array_combine() with array_map() to add -- prefix to the $route key. Refer from https://stackoverflow.com/a/2609278/128761
305:                         $arguments = array_combine(array_map(function($k) {return '--'.$k;}, array_keys($route)), $route);
306:                         $arguments = array_merge($arguments, [
307:                             'command' => 'system\route',
308:                             'act' => 'add',
309:                         ]);
310:                         $ArrayInput = new ArrayInput($arguments);
311:                         $Command->run($ArrayInput, new NullOutput());
312:                         unset($arguments, $ArrayInput);
313:                         // end call other command.
314:                     }// endforeach;
315:                     unset($Command, $route, $RoutesDb);
316:                 }// endif $routes
317: 
318:                 if (
319:                     is_file(MODULE_PATH . DIRECTORY_SEPARATOR . $options['module_system_name'] . DIRECTORY_SEPARATOR . 'composer.json') && 
320:                     is_file(ROOT_PATH . DIRECTORY_SEPARATOR . 'composer.json')
321:                 ) {
322:                     // if composer.json found in module's folder and app's root folder.
323:                     // copy the "require" and "require-dev" to app's composer.json.

437:                             $this->copyEnabledModulesComposer();
438:                             // finish delete and copy composer.json

440:                         } else {
