107:         $output = '<?php'."\n";

111:         $output .= ' */'."\n";
112:         $output .= "\n";

114:         if (is_array($result)) {

116:                 $output .= '    // '.str_replace(array("\r\n", "\r", "\n"), '', $row->config_description)."\n";
117:                 $output .= '    \''.$row->config_name.'\' => ['."\n";
118:                 $output .= '        \'config_value\' => \''.str_replace('\'', '\\\'', $row->config_value).'\','."\n";
119:                 $output .= '        \'config_core\' => '.($row->config_core == '1' ? 'true' : 'false').','."\n";
120:                 $output .= '    ],'."\n";
121:             }// endforeach;

249:     /**
250:      * Get the path to config file.<br>
251:      * The configurations are in db inside config table and it will be auto generate into config file for quick access.
252:      * 
253:      * @return string Return the path to config file.
254:      */
255:     protected function getConfigFile()
256:     {
257:         if (is_file($this->configDbFolder . DIRECTORY_SEPARATOR . $this->configDbFile)) {
258:             // if it was built, check how old is it.
259:             $FileSystem = new \System\Libraries\FileSystem($this->configDbFolder);
260:             $fileTs = $FileSystem->getTimestamp($this->configDbFile);
261:             $dayOld = ((time()-$fileTs)/60/60/24);
262:             unset($fileTs);
263: 
264:             if ($dayOld > 30) {
265:                 // older than 30 days, rebuild config.
266:                 $this->buildConfigFile();
267:             }
268: 
269:             unset($dayOld, $FileSystem);
270:             return $this->configDbFolder . DIRECTORY_SEPARATOR . $this->configDbFile;
271:         } elseif ($this->buildConfigFile() === true) {
272:             // if never build the configuration file before, build it.
273:             return $this->configDbFolder . DIRECTORY_SEPARATOR . $this->configDbFile;
274:         } else {
275:             return null;
276:         }
277:     }// getConfigFile
