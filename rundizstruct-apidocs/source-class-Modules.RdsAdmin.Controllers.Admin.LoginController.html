291:     /**
292:      * Response XMLHttpRequest based on request's accept header.
293:      * 
294:      * Example: request's accept header is application/json then it will return application/json.<br>
295:      * If accept header is application/xml or text/xml then it will return application/xml.<br>
296:      * Otherwise it will return application/json.
297:      * 
298:      * The data that will be response must be set to `$data` property.
299:      * 
300:      * @param ServerRequestInterface $request
301:      * @param ResponseInterface $response
302:      * @return ResponseInterface
303:      */
304:     private function responseXhr(ServerRequestInterface $request, ResponseInterface $response)
305:     {
306:         $httpAccept = $request->getHeaderLine('Accept');
307:         if (stripos($httpAccept, 'application/json') !== false) {
308:             $httpAccept = 'application/json';
309:         } elseif (
310:             (
311:                 stripos($httpAccept, 'application/xml') !== false || stripos($httpAccept, 'text/xml') !== false
312:             ) &&
313:             stripos($httpAccept, 'text/html') === false
314:         ) {
315:             $httpAccept = 'application/xml';
316:         } elseif (
317:             $httpAccept === 'application/x-yaml' ||
318:             $httpAccept === 'application/yaml' ||
319:             $httpAccept === 'text/x-yaml' ||
320:             $httpAccept === 'text/yaml'
321:         ) {
322:             $httpAccept = 'application/x-yaml';
323:         } else {
324:             $httpAccept = 'text/html';
325:         }
326: 
327:         // no cache.
328:         $response = $response
329:             ->withHeader('Cache-Control', 'no-store, no-cache, must-revalidate, max-age=0')
330:             ->withAddedHeader('Cache-Control', 'post-check=0, pre-check=0')
331:             ->withHeader('Pragma', 'no-cache')
332:             ->withHeader('Expires', '0');
333: 
334:         switch ($httpAccept) {
335:             case 'application/xml':
336:                 $response = $response->withHeader('Content-Type', 'application/xml;charset=utf-8');
337:                 $SimpleXml = new \SimpleXMLElement('<?xml version="1.0"?><data></data>');
338:                 $Xml = new \System\Libraries\Xml();

340:                 $content = $SimpleXml->asXML();
341:                 unset($SimpleXml, $Xml);
342:                 return $response->write($content);

344:             case 'application/json':
345:             default:
