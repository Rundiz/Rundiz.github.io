 13: /**
 14:  * PSR-4 auto load
 15:  * 
 16:  * An example of a general-purpose implementation that includes the optional
 17:  * functionality of allowing multiple base directories for a single namespace
 18:  * prefix.
 19:  *
 20:  * Given a foo-bar package of classes in the file system at the following
 21:  * paths ...
 22:  *<pre>
 23:  *     /path/to/packages/foo-bar/
 24:  *         src/
 25:  *             Baz.php             # Foo\Bar\Baz
 26:  *             Qux/
 27:  *                 Quux.php        # Foo\Bar\Qux\Quux
 28:  *         tests/
 29:  *             BazTest.php         # Foo\Bar\BazTest
 30:  *             Qux/
 31:  *                 QuuxTest.php    # Foo\Bar\Qux\QuuxTest
 32:  *</pre>
 33:  * ... add the path to the class files for the \Foo\Bar\ namespace prefix
 34:  * as follows:
 35:  *<pre>
 36:  *      <?php
 37:  *      // instantiate the loader
 38:  *      $loader = new \Example\Psr4AutoloaderClass;
 39:  *
 40:  *      // register the autoloader
 41:  *      $loader->register();
 42:  *
 43:  *      // register the base directories for the namespace prefix
 44:  *      $loader->addNamespace('Foo\Bar', '/path/to/packages/foo-bar/src');
 45:  *      $loader->addNamespace('Foo\Bar', '/path/to/packages/foo-bar/tests');
 46:  *</pre>
 47:  * The following line would cause the autoloader to attempt to load the
 48:  * \Foo\Bar\Qux\Quux class from /path/to/packages/foo-bar/src/Qux/Quux.php:
 49:  *<pre>
 50:  *      <?php
 51:  *      new \Foo\Bar\Qux\Quux;
 52:  *</pre>
 53:  * The following line would cause the autoloader to attempt to load the
 54:  * \Foo\Bar\Qux\QuuxTest class from /path/to/packages/foo-bar/tests/Qux/QuuxTest.php:
 55:  *<pre>
 56:  *      <?php
 57:  *      new \Foo\Bar\Qux\QuuxTest;
 58:  *</pre>
 59:  */
 60: class Autoload
 61: {
 62:     /**
 63:      * An associative array where the key is a namespace prefix and the value
 64:      * is an array of base directories for classes in that namespace.
 65:      *
 66:      * @var array
 67:      */
 68:     protected $prefixes = array();
 69: 
 70:     /**
 71:      * Register loader with SPL autoloader stack.
 72:      *
 73:      * @return void
 74:      */
 75:     public function register()
 76:     {
 77:         spl_autoload_register(array($this, 'loadClass'));
 78:     }
 79: 
 80: 
 81:     /**
 82:      * Adds a base directory for a namespace prefix.
 83:      *
 84:      * @param string $prefix The namespace prefix.
 85:      * @param string $base_dir A base directory for class files in the
 86:      * namespace.
 87:      * @param bool $prepend If true, prepend the base directory to the stack
 88:      * instead of appending it; this causes it to be searched first rather
 89:      * than last.
 90:      * @return void
 91:      */
 92:     public function addNamespace($prefix, $base_dir, $prepend = false)
 93:     {
 94:         // normalize namespace prefix
 95:         $prefix = trim($prefix, '\\') . '\\';
 96: 
 97:         // normalize the base directory with a trailing separator
 98:         $base_dir = rtrim($base_dir, DIRECTORY_SEPARATOR) . '/';
 99: 
100:         // initialize the namespace prefix array
101:         if (isset($this->prefixes[$prefix]) === false) {
102:             $this->prefixes[$prefix] = array();
103:         }
104: 
105:         // retain the base directory for the namespace prefix
106:         if ($prepend) {
107:             array_unshift($this->prefixes[$prefix], $base_dir);
108:         } else {
109:             array_push($this->prefixes[$prefix], $base_dir);
110:         }
111:     }
112: 
113: 
114:     /**
115:      * Loads the class file for a given class name.
116:      *
117:      * @param string $class The fully-qualified class name.
118:      * @return mixed The mapped file name on success, or boolean false on
119:      * failure.
120:      */
121:     public function loadClass($class)
122:     {
123:         // the current namespace prefix
124:         $prefix = $class;
125: 
126:         // work backwards through the namespace names of the fully-qualified
127:         // class name to find a mapped file name
128:         while (false !== $pos = strrpos($prefix, '\\')) {
129: 
130:             // retain the trailing namespace separator in the prefix
131:             $prefix = substr($class, 0, $pos + 1);
132: 
133:             // the rest is the relative class name
134:             $relative_class = substr($class, $pos + 1);
135: 
136:             // try to load a mapped file for the prefix and relative class
137:             $mapped_file = $this->loadMappedFile($prefix, $relative_class);
138:             if ($mapped_file) {
139:                 return $mapped_file;
140:             }
141: 
142:             // remove the trailing namespace separator for the next iteration
143:             // of strrpos()
144:             $prefix = rtrim($prefix, '\\');
145:         }
146: 
147:         // never found a mapped file
148:         return false;
149:     }
150: 
151: 
152:     /**
153:      * Load the mapped file for a namespace prefix and relative class.
154:      *
155:      * @param string $prefix The namespace prefix.
156:      * @param string $relative_class The relative class name.
157:      * @return mixed Boolean false if no mapped file can be loaded, or the
158:      * name of the mapped file that was loaded.
159:      */
160:     protected function loadMappedFile($prefix, $relative_class)
161:     {
162:         // are there any base directories for this namespace prefix?
163:         if (isset($this->prefixes[$prefix]) === false) {
164:             return false;
165:         }
166: 
167:         // look through base directories for this namespace prefix
168:         foreach ($this->prefixes[$prefix] as $base_dir) {
169: 
170:             // replace the namespace prefix with the base directory,
171:             // replace namespace separators with directory separators
172:             // in the relative class name, append with .php
173:             $file = $base_dir
174:                   . str_replace('\\', '/', $relative_class)
175:                   . '.php';
176: 
177:             // if the mapped file exists, require it
178:             if ($this->requireFile($file)) {
179:                 // yes, we're done
180:                 return $file;
181:             }
182:         }
183: 
184:         // never found it
185:         return false;
186:     }
187: 
188: 
189:     /**
190:      * If a file exists, require it from the file system.
191:      *
192:      * @param string $file The file to require.
193:      * @return bool True if the file exists, false if not.
194:      */
195:     protected function requireFile($file)
196:     {
197:         if (file_exists($file)) {
198:             require $file;
199:             return true;
200:         }
201:         return false;
202:     }
